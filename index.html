// server.js const http = require('http'); const htmlContent = ` Homework 1

I did my homework!

`; const server = http.createServer((req, res) => { res.writeHead(200, { 'Content-Type': 'text/html' }); res.end(htmlContent); }); const PORT = process.env.PORT || 3000; server.listen(PORT, () => { console.log(`Server is running at http://localhost:${PORT}/`); });
